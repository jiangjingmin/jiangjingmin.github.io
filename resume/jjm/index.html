# 1. 截取字符串的三种方式

  • str.slice(start,end) (opens new window) —— 推荐使用
  • str.substring(start,stop) (opens new window)
  • str.substr(start,length) (opens new window) —— 反对使用
let str = "abcdef";
// 0
str = str.slice(0); // abcdef
str = str.substring(0); // abcdef
str = str.substr(0); // abcdef
// 使用一个参数
str = str.slice(2); // cdef
str = str.substring(2); // cdef
str = str.substr(2); // cdef

// 使用两个参数
str = str.slice(2, 4); // cd
str = str.substring(2, 4); // cd
str = str.substr(2, 4); // cdef

// 使用两个负数
str = str.slice(1, -3); // bc
str = str.substring(1, -3); // a #负数转换为0
str = str.substr(1, -3); // ' ' #负数转换为0

// 注意：substring() ：如果参数 start 与 stop 相等，那么该方法返回的就是一个空串（即长度为 0 的字符串）。如果 start 比 stop 大，那么该方法在提取子串之前会先交换这两个参数。

# 2. js 去重

function unique(arr) {
  return arr.filter(function(item, index, arr) {
    //当前元素，在原始数组中的第一个索引==当前索引值，否则返回当前元素
    return arr.indexOf(item, 0) === index;
  });
}
