# 3. 数组的最大深度

// 获取数组的最大深度
const arr = [1, [2, [3], 4], [5], [6], [7], [8], 9];

function getMaxArrayDepth(arr) {
    let depth = 0; // 记录数组深度
    
    function setDepth (newArr) {
        // 不是数组，则返回
        if(!Array.isArray(newArr)) return;
        
        if(!Array.isArray(newArr)) return;

        for (let i=0;i< newArr.length;i++) {
            setDepth(newArr[i]);
        }
    }
    setDepth(arr);
    return depth;
}
const depth = getMaxArrayDepth(arr);
console.log(depth);

# 笔试：3. 使用promise加载一个网络图片，如果失败，返回一张默认图片占位

const loadImage = (url, defaultImg) => {
    const image = new Image();
    return new Promise((resolve, reject) => {
        images.src = url;
        document.body.appendChild(image);
        image.onload = () => {
            resolve('load images success');
        }
        image.onerror = () => {
            image.src = defaultImg;
            reject('load images success');
        }
    })
}

loadImage("").then(res => {
    console.log(res);
}).catch(error => {
    console.log(error);
})

export default loadImage;
